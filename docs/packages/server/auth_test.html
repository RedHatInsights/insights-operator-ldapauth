func
 
TestServerLoginUnauthorized
(
t
 
*
testing
.
T
)
 
{

	
body
,
 
_
 
:=
 
json
.
Marshal
(
map
[
string
]
interface
{
}
{
"login"
:
 
"tester"
,
 
"password"
:
 
"1234"
}
)

	
req
,
 
err
 
:=
 
http
.
NewRequest
(
"POST"
,
 
APIPrefix
+
"login"
,
 
bytes
.
NewBuffer
(
body
)
)

	
if
 
err
 
!=
 
nil
 
{

		
t
.
Errorf
(
"Unexpected error %s"
,
 
err
)

	
}

	
rr
 
:=
 
httptest
.
NewRecorder
(
)

	
handler
 
:=
 
http
.
HandlerFunc
(
testServer
.

)

	
handler
.
ServeHTTP
(
rr
,
 
req
)

Check content-type header

 
	
if
 
contentType
 
:=
 
rr
.
Header
(
)
.
Get
(
"Content-Type"
)
;
 
contentType
 
!=
 
"application/json; charset=utf-8"
 
{

		
t
.
Errorf
(
"Expected Content-Type header value: %v but got %v"
,

			
"application/json; charset=utf-8"
,
 
contentType
)

	
}

Check the status code

 
	
if
 
status
 
:=
 
rr
.
Code
;
 
status
 
!=
 
http
.
StatusUnauthorized
 
{

		
t
.
Errorf
(
"Expected status code: %v but got %v"
,

			
http
.
StatusUnauthorized
,
 
status
)

	
}



}
